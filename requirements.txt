# STEP 0. conda 환경 생성
#   conda create -n p310_pj_styleflow python=3.10
#   conda activate p310_pj_styleflow
#
# STEP 1. dlib (conda-forge 전용, Visual C++ 없이 설치 불가)
#   conda install -c conda-forge dlib=19.24.1
#
# STEP 2. setuptools 버전 고정 (CLIP 설치 전 필수)
#   pip install setuptools==69.5.1
#
# STEP 3. 기본 패키지 설치 (torch/CLIP 제외)
#   pip install -r requirements.txt
#
# STEP 4. tensorflow-intel 설치
#   pip install tensorflow-intel==2.13.0
#   ※ typing_extensions가 4.5.0으로 다운그레이드됨 → STEP 5에서 복원
#
# STEP 5. mediapipe 설치
#   pip install mediapipe==0.10.9
#   ※ protobuf가 3.20.3으로 다운그레이드됨 (TF도 허용하는 버전이라 정상)
#
# STEP 6. typing_extensions 복원 (TF가 4.5.0으로 내려버리기 때문)
#   pip install typing_extensions==4.10.0
#
# STEP 7. CLIP 설치 (--no-build-isolation 필수)
#   pip install git+https://github.com/openai/CLIP.git@a1d071733d7111c9c014f024669f959182114e33 --no-build-isolation
#
# STEP 8. torch/torchvision 마지막 설치 (--no-deps 필수, 구버전 의존성 방지)
#   [CPU 로컬]
#   pip install torch==1.13.1 torchvision==0.14.1 --index-url https://download.pytorch.org/whl/cpu --no-deps
#   [GPU 서버 — CUDA 11.7 기준]
#   pip install torch==1.13.1+cu117 torchvision==0.14.1+cu117 --index-url https://download.pytorch.org/whl/cu117 --no-deps
#
# ============================================================
# [알려진 버전 충돌 — 런타임에서는 정상 동작]
#   - tensorflow-intel 2.13.0: typing_extensions<4.6.0 요구
#     → 실제론 4.10.0에서도 정상 작동 확인됨
#   - mediapipe 0.10.9: protobuf<4 요구
#     → protobuf 3.20.3으로 다운그레이드됨, TF도 해당 버전 허용
#   - transformers==4.44.2: huggingface-hub<1.0 런타임 체크 요구
#     → 실제 API 호환성 문제 없음. 설치 후 아래 파일에서 "huggingface-hub" 항목 제거 필요:
#       anaconda3/envs/p310_pj_styleflow/Lib/site-packages/transformers/dependency_versions_check.py
#       pkgs_to_check_at_runtime 리스트에서 "huggingface-hub" 줄 삭제
# ============================================================
# 주의: dlib, torch, torchvision, clip은 아래 목록에서 제외됨
